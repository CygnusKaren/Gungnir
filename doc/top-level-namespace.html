def loadDSL
  # DSLファイルを読み込む
  # 作業ディレクトリ側の DSL を読む
  dsl_file = File.join(Dir.pwd, 'project_dsl.rb')
    unless File.exist?(dsl_file)
    abort "project_dsl.rb が見つかりません: #{dsl_file}"
  end
  Object.send(:remove_const, :Gungnir_DSL) if defined?(Gungnir_DSL)
  load dsl_file  
  @file_to_paks = {} 
end
