def initialize(name, &block)
  @name = name
  @paks = []
  @common_files = []
  @export_path = nil
  instance_eval(&block) if block_given?
end

def pakEach(list, prefix:, dir: "./", &block)
  list.each do |v|
    pakName "#{prefix}-#{v}" do
      instance_exec(v, &block)
    end
  end
end

def pakName(name, dir: "./", &block)
  pak = Pak.new(name)
  pak.instance_eval(&block) if block_given?
  @paks << pak
end
